prelude.Boolean

Определение

type Boolean = true | false with
    open comparable
    ...

Модуль, содержащий функции сравнения на основе оператора <=>

Операторы

<=>

let <=> this other =
    ...

Реализация модуля comparable.Производит сравнение двух значений.

print (true <=> false) //-> -1

not

let not this =
    match this
    | true = false
    | false = true

Логическое НЕ.

print not true  //-> false
print not false //-> true

and

let and this other =
    match this
    | true  = match other
              | true = true
              | false = false
    | false = false

Логическое И.

print true and true   //-> true
print true and false  //-> false
print false and true  //-> false
print false and false //-> false

or

let or this other =
    match this
    | true  = true
    | false = match other
              | true = true
              | false = false

Логическое ИЛИ.

print true or true   //-> true
print true or false  //-> true
print false or true  //-> true
print false or false //-> false

xor

let xor this other =
    match this
    | true  = match other
              | true = false
              | false = true
    | false = match other
              | true = true
              | false = false

Логическое ИЛИ.

print true xor true   //-> false
print true xor false  //-> true
print false xor true  //-> true
print false xor false //-> false

Функции

String

let String (this: Boolean): String =
    match this
    | true = "true"
    | false = "false"

Производит сравнение двух значений.

print (5.compare 6) //-> -1

Number

let Number (this: Boolean): Number =
    match this
    | true = 1
    | false = 0

Возвращает минимальное из двух значений.

print (5.min 6) //-> 5
