Ord

This class contains functions for comparsion.
type Ord = class
    ...
type NumWrap = NumWrap a
    implements Ord
    // Implementation of Ord mixin
    let compare y x =
        if x.a < y.a:
            return -1
        if x.a > y.a:
            return 1
        return 0

let w1 = NumWrap 0
let w2 = NumWrap 7

println (w1 < w2)               //-> true
println (w1 > w2)              //-> false
println (w1 >= w2)            //-> false
println (w1.min w2)          //-> 0
println (w1.max w2)         //-> 7
println (w1.compare w2) //-> -1
