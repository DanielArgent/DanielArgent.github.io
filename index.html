# Insertion sort for all list
def List.Insort(x) do
    i = 1
    while i < *x do
        if x[i] < x[i-1] do
            prom = x[i]
            x[i] = x[i-1]
            x[i-1] = prom
            i = 1
        end else i++
    end 
    return x
end
Test: 
puts [4, 2, 5, 7].Insort() #=> [2, 4, 5, 7]
        
        
    	    
# Ханойские башни
def Hanoi(n, f, t, w) do
    if n == 0 
        return null
    self(n - 1, f, w, t)
    # From f to t
    puts f + "->" + t
    self(n - 1, w, t, f)
end
const n = 8 # Количество дисков
Hanoi(n, 1, 2, 3)
        
       
    	    
# Class Greeter
class Greeter(name) do
    def Init(this)
        this.name = this.name.ToUpp()

    def salute(this)
        puts $"Hello {this.name}"
end

# Create a new object
g = Greeter("world")

# Result is "Hello WORLD!"
g.salute()
